回顾JS函数的方法

Created2022-10-09|Updated2022-10-09
|Post View:
<script>
  let obj = {
    name: "hello",
  };
  function fun(...args) {
    console.log(this);
    console.log(args);
  }
  // 函数这几个方法的用法多用于改变函数的this指向  和方法之间的继承 也可以是多个方法的继承
  // call，apply这两个方法都是以原函数为基础不改变原函数，而bind方法是直接创建了一个新函数
  // call方法第一个参数为改变this指向的对象，第二个参数为原函数的实参如果要多个参数的话逗号相隔即可(通过列表依次传参)
  const a = fun.call(document, 1, 2, 3);
  //  apply方法与call方法一样，不同的地方在于第二个参数位置传的是以数组的形式
  const b = fun.apply(obj, [4, 5, 6]);
  // bind方法不会直接调用函数而是返回一个新的函数需要重新()来调用才行
  const c = fun.bind(null, 7, 8, 9)();
</script>
